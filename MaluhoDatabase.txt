create database therealmaluho; 
use therealmaluho;
create table user(userid int not null auto_increment, lastname varchar(255)not null,firstname varchar(255)not null,username varchar(255) not null,pass varchar(255) not null, access varchar(50) not null, securityquestion varchar(255) not null, answer varchar(255) not null, primary key(userid) );
create table product(Productid int not null auto_increment, Apparel varchar(255) not null,Brand varchar(255)not null, Name varchar(255) not null, primary key(ProductID));
create table inventory(Inventoryid int not null auto_increment, Barcode varchar(255)not null,Apparel varchar(255)not null,Brand varchar(255)not null,ItemName varchar(255)not null,Size varchar(255)not null,Quantity varchar(255)not null,Itemcost varchar(255)not null, Note varchar(255)not null, Date varchar(255)not null, primary key(Inventoryid) );
create table customer(Id int not null auto_increment,Name varchar(255)not null,Location varchar(255)not null,PhoneNumber varchar(255)not null,InventoryID int not null,Brand varchar(255)not null,ItemName varchar(255)not null,Cost varchar(255)not null,Price varchar(255)not null,Status varchar(255)not null,Miner int not null,Date varchar(255)not null, primary key(Id));
create table timesheet (timesheetID int not null auto_increment,lastname varchar(255)not null,firstname varchar(255)not null,userid int(50)not null,Tag varchar(255)not null,TimeIn varchar(255)not null,BreakIn varchar(255) not null, BreakOut varchar(255)not null,TimeOut varchar(255) not null,TotalHours varchar(255) not null,Date varchar (255) not null, primary key(timesheetID));
create table storecustomer(CustomerID int not null auto_increment,Barcode varchar(255) not null, Apparel varchar (255)not null, Brand varchar(255) not null, ItemName varchar(255) not null, Cost varchar (255)not null, Price varchar(255)not null, primary key (CustomerID));
